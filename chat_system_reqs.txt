function handleDownload_(token) {
  const props = PropertiesService.getScriptProperties();
  const secret = props.getProperty("DOWNLOAD_SECRET");
  const decoded = Utilities.base64DecodeWebSafe(token);
  const payload = decoded.join("|");

  const [book_id, asset_type, email_hash, expires_at, sig] = payload.split("|");

  if (Date.now() > Number(expires_at)) {
    return ContentService.createTextOutput("Link expired");
  }

  const expected = Utilities.computeHmacSha256Signature(
    payload.replace("|" + sig, ""),
    secret
  );

  if (Utilities.base64EncodeWebSafe(expected) !== sig) {
    return ContentService.createTextOutput("Invalid link");
  }

  const base = props.getProperty("DOWNLOAD_BASE_URL");
  const path =
    asset_type === "companion"
      ? `/companions/${book_id}.pdf`
      : `/resources/${book_id}.pdf`;

  return HtmlService.createHtmlOutput(
    `<meta http-equiv="refresh" content="0; url=${base}${path}">`
  );
}

function weeklyKpiExporter() {
  const ss = SpreadsheetApp.getActive();
  const events = ss.getSheetByName("events");
  if (!events) return;

  let out = ss.getSheetByName("weekly_kpis");
  if (!out) {
    out = ss.insertSheet("weekly_kpis");
    out.appendRow([
      "week_start",
      "brand_id",
      "captures",
      "downloads",
      "followups_sent"
    ]);
  }

  const rows = events.getDataRange().getValues();
  const header = rows[0];
  const idx = Object.fromEntries(header.map((h, i) => [h, i]));

  const now = new Date();
  const weekStart = new Date(now);
  weekStart.setDate(now.getDate() - now.getDay()); // Sunday start
  weekStart.setHours(0, 0, 0, 0);

  const counts = {};

  for (let i = 1; i < rows.length; i++) {
    const eventType = rows[i][idx.event_type];
    const brandId = rows[i][idx.brand_id];
    const createdAt = new Date(rows[i][idx.created_at]);

    if (createdAt < weekStart) continue;

    if (!counts[brandId]) {
      counts[brandId] = { capture: 0, download: 0, followup_sent: 0 };
    }

    if (eventType === "capture") counts[brandId].capture++;
    if (eventType === "download") counts[brandId].download++;
    if (eventType === "followup_sent") counts[brandId].followup_sent++;
  }

  Object.entries(counts).forEach(([brandId, c]) => {
    out.appendRow([
      weekStart,
      brandId,
      c.capture,
      c.download,
      c.followup_sent
    ]);
  });
}

from __future__ import annotations

from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple
import re
import yaml


@dataclass(frozen=True)
class ProseViolation:
    rule_id: str
    severity: str  # "FAIL" | "WARN"
    phrase: str
    chapter_num: Optional[int]
    section_id: Optional[str]
    start_offset: int
    snippet: str


def _load_yaml(path: Path) -> Dict[str, Any]:
    with path.open("r", encoding="utf-8") as f:
        return yaml.safe_load(f)


def _compile_phrases(phrases: List[str]) -> List[Tuple[str, re.Pattern]]:
    out = []
    for p in phrases:
        # phrase-level literal match, case-insensitive
        pat = re.compile(re.escape(p), re.IGNORECASE)
        out.append((p, pat))
    return out


def _build_snippet(text: str, start: int, width: int = 90) -> str:
    lo = max(0, start - width // 2)
    hi = min(len(text), start + width // 2)
    s = text[lo:hi].replace("\n", " ")
    return s.strip()


def run_prose_guard(
    rendered_text: str,
    format_type: str,
    template_family: str,
    *,
    strict: bool = True,
    chapter_num: Optional[int] = None,
    section_id: Optional[str] = None,
    policy_path: Path = Path("phoenix_v4/policy/order_implication_phrases.yaml"),
) -> List[ProseViolation]:
    """
    Universal prose QA for ALL assembly modes.
    - strict=True: FAIL on forbid + WARN on warn
    - strict=False: WARN only
    """
    policy = _load_yaml(policy_path)
    forbid = policy.get("order_implication_forbid", [])
    warn = policy.get("order_implication_warn", [])

    forbid_compiled = _compile_phrases(forbid)
    warn_compiled = _compile_phrases(warn)

    violations: List[ProseViolation] = []

    def _scan(rule_id: str, severity: str, compiled: List[Tuple[str, re.Pattern]]):
        for phrase, pat in compiled:
            for m in pat.finditer(rendered_text):
                violations.append(
                    ProseViolation(
                        rule_id=rule_id,
                        severity=severity,
                        phrase=phrase,
                        chapter_num=chapter_num,
                        section_id=section_id,
                        start_offset=m.start(),
                        snippet=_build_snippet(rendered_text, m.start()),
                    )
                )

    if strict:
        _scan("order_implication_forbid", "FAIL", forbid_compiled)
        _scan("order_implication_warn", "WARN", warn_compiled)
    else:
        _scan("order_implication_forbid", "WARN", forbid_compiled)
        _scan("order_implication_warn", "WARN", warn_compiled)

    return violations


def violations_to_dict(vs: List[ProseViolation]) -> List[Dict[str, Any]]:
    return [asdict(v) for v in vs]

#!/usr/bin/env python3
import argparse
import json
from collections import Counter, defaultdict
from pathlib import Path


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--jsonl", default="artifacts/v4_1000_book_qa_report.jsonl")
    ap.add_argument("--out-json", default="artifacts/v4_1000_book_qa_report.json")
    ap.add_argument("--out-md", default="artifacts/v4_1000_book_qa_report.md")
    args = ap.parse_args()

    p = Path(args.jsonl)
    rows = []
    if p.exists():
        with p.open("r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if line:
                    rows.append(json.loads(line))

    total = len(rows)
    passed = sum(1 for r in rows if r.get("pass") is True)
    failed = total - passed

    by_rule = Counter()
    by_family = Counter()
    by_format = Counter()

    for r in rows:
        fam = r.get("template_family", "unknown")
        fmt = r.get("format_type", "unknown")
        if not r.get("pass", False):
            by_family[fam] += 1
            by_format[fmt] += 1
            for v in r.get("violations", []):
                by_rule[v.get("rule_id", "unknown")] += 1

    out_json = {
        "total_books": total,
        "passed": passed,
        "failed": failed,
        "pass_rate": (passed / total) if total else 0.0,
        "failures_by_rule": dict(by_rule),
        "failures_by_template_family": dict(by_family),
        "failures_by_format_type": dict(by_format),
    }

    Path(args.out_json).write_text(json.dumps(out_json, indent=2), encoding="utf-8")

    md = []
    md.append("# V4 1000-Book QA Report")
    md.append("")
    md.append(f"- Total: {total}")
    md.append(f"- Passed: {passed}")
    md.append(f"- Failed: {failed}")
    md.append(f"- Pass rate: {out_json['pass_rate']:.2%}")
    md.append("")
    md.append("## Failures by rule")
    for k, v in by_rule.most_common():
        md.append(f"- {k}: {v}")
    md.append("")
    md.append("## Failures by template family")
    for k, v in by_family.most_common():
        md.append(f"- {k}: {v}")
    md.append("")
    md.append("## Failures by format type")
    for k, v in by_format.most_common():
        md.append(f"- {k}: {v}")

    Path(args.out_md).write_text("\n".join(md) + "\n", encoding="utf-8")


if __name__ == "__main__":
    main()

prose_guard.py skeleton (single entrypoint for Gate Stack)

from __future__ import annotations

from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple
import re
import yaml

# ---- Violation model (triage-ready) ----

@dataclass(frozen=True)
class Violation:
    gate: str
    rule_id: str
    severity: str               # "FAIL" | "WARN"
    chapter_num: Optional[int]
    section_id: Optional[str]
    snippet: str
    start_offset: Optional[int]
    meta: Dict[str, Any]

def _vdict(v: Violation) -> Dict[str, Any]:
    return asdict(v)

# ---- Loaders ----

def _load_yaml(path: Path) -> Dict[str, Any]:
    with path.open("r", encoding="utf-8") as f:
        return yaml.safe_load(f)

def _compile_literal_phrases(phrases: List[str]) -> List[Tuple[str, re.Pattern]]:
    return [(p, re.compile(re.escape(p), re.IGNORECASE)) for p in phrases]

def _snippet(text: str, start: int, width: int = 120) -> str:
    lo = max(0, start - width // 2)
    hi = min(len(text), start + width // 2)
    return text[lo:hi].replace("\n", " ").strip()

# ---- Gate #1: Order-implication + banned_patterns ----

def _run_order_implication_gate(
    *,
    rendered_text: str,
    format_type: str,
    template_family: str,
    strict: bool,
    order_dependent: bool,
    policy_path: Path,
    format_rules_path: Path,
) -> List[Violation]:
    violations: List[Violation] = []

    policy = _load_yaml(policy_path)
    forbid = policy.get("order_implication_forbid", [])
    warn = policy.get("order_implication_warn", [])

    # If family is order-dependent, default behavior: warn-only (no hard fails)
    effective_strict = strict and (not order_dependent)
    forbid_severity = "FAIL" if effective_strict else "WARN"

    for phrase, pat in _compile_literal_phrases(forbid):
        for m in pat.finditer(rendered_text):
            violations.append(Violation(
                gate="GATE_1_ORDER_IMPLICATION",
                rule_id="order_implication_forbid",
                severity=forbid_severity,
                chapter_num=None,
                section_id=None,
                snippet=_snippet(rendered_text, m.start()),
                start_offset=m.start(),
                meta={"phrase": phrase, "format_type": format_type, "template_family": template_family},
            ))

    for phrase, pat in _compile_literal_phrases(warn):
        for m in pat.finditer(rendered_text):
            violations.append(Violation(
                gate="GATE_1_ORDER_IMPLICATION",
                rule_id="order_implication_warn",
                severity="WARN",
                chapter_num=None,
                section_id=None,
                snippet=_snippet(rendered_text, m.start()),
                start_offset=m.start(),
                meta={"phrase": phrase, "format_type": format_type, "template_family": template_family},
            ))

    # Apply format_rules banned_patterns with default fallback
    fmt_rules = _load_yaml(format_rules_path)
    defaults = fmt_rules.get("defaults", {})
    formats = fmt_rules.get("formats", {})

    # baseline banned patterns fallback:
    banned = []
    if format_type in formats:
        banned = formats[format_type].get("banned_patterns") or []
    if not banned:
        # fallback: if defaults specify a ref, use policy forbid list; otherwise copy from standard_book if present
        ref = defaults.get("banned_patterns_ref")
        if ref == "order_implication_forbid":
            banned = forbid
        elif "standard_book" in formats:
            banned = formats["standard_book"].get("banned_patterns") or forbid
        else:
            banned = forbid

    for phrase, pat in _compile_literal_phrases(banned):
        for m in pat.finditer(rendered_text):
            # banned_patterns are treated as FAIL in strict mode
            sev = "FAIL" if strict else "WARN"
            violations.append(Violation(
                gate="GATE_1_ORDER_IMPLICATION",
                rule_id="format_banned_patterns",
                severity=sev,
                chapter_num=None,
                section_id=None,
                snippet=_snippet(rendered_text, m.start()),
                start_offset=m.start(),
                meta={"phrase": phrase, "format_type": format_type, "template_family": template_family},
            ))

    return violations

What it is
Omega V4 TTS turns an assembled V4 book into audio using ElevenLabs. It uses the same rendering pipeline as Golden Phoenix: SOURCE_OF_TRUTH/tools/atoms_render.py (voice config, SSML chunking, caching, retries, optional fail-open mock).
Requirements
ELEVENLABS_API_KEY in the environment (needed for real TTS).
If the key is missing and fail_open=True (default), the pipeline can still run and use mock audio instead of failing.
CLI
From repo root:
# Assemble from plan and render to MP3s
python3 -m phoenix_v4.tts.omega_tts phoenix_v4/plans/<plan>.yaml -o /path/to/output

# Override voice
python3 -m phoenix_v4.tts.omega_tts phoenix_v4/plans/<plan>.yaml -o /path/to/output --voice-id YOUR_ELEVENLABS_VOICE_ID

# Strict mode: raise on TTS errors (no mock)
python3 -m phoenix_v4.tts.omega_tts phoenix_v4/plans/<plan>.yaml -o /path/to/output --no-fail-open -v

Arguments
Argument	Purpose
plan_path	V4 plan YAML (book is assembled from this, then rendered).
-o / --output-dir	Directory for MP3s and manifest (required).
--voice-id	ElevenLabs voice ID; overrides persona-based voice.
--no-fail-open	On TTS/API errors, raise instead of using mock audio.
-v / --verbose	More logging.
--repo-root	Repo root for config (default: auto-detect).
Workflow (what the CLI does)
Assemble
Load the plan and run assemble_book_v4(plan_path, sections_root, repo_root=repo_root) to produce a Book (front matter, chapters, back matter).
Segment
Split the book into TTS segments:
front_matter (if present)
ch_01, ch_02, … (one per chapter with content)
back_matter (if present)
Voice
Resolve the ElevenLabs voice ID (see order below). If --voice-id is set, it wins.
SSML
For each segment, convert plain text to SSML (e.g. sentence/paragraph structure) via _prose_to_ssml() for chunking.
Render
For each segment, call atoms_render.render_ssml_with_chunking():
Uses create_voice_config(voice_id) and the shared pipeline (chunking, retries, optional fail-open mock).
Writes one MP3 per segment: front_matter.mp3, ch_01.mp3, …, back_matter.mp3.
Manifest
Write omega_tts_manifest.json in the output dir with book_id, voice_id, persona, and the list of segment IDs and MP3 paths.
Voice resolution order
--voice-id (CLI) or voice_id_override (API)
unified_personas.yaml → personas.<persona>.voice_profile_id
SOURCE_OF_TRUTH/config/persona_voice_registry.json → persona_voice_mappings.<persona>.voice_profile_id or top-level persona
SOURCE_OF_TRUTH/config/voice_id_map.json if the value from 2/3 is a symbolic key (e.g. gen_alpha) → raw ElevenLabs ID
Default: Luna voice ID fDeOZu1sNd7qahm2fV4k
Output
MP3s: front_matter.mp3, ch_01.mp3, ch_02.mp3, …, back_matter.mp3 in the given output directory.
Manifest: omega_tts_manifest.json with book_id, voice_id, persona, and segments (segment_id + path per segment).
API (no CLI)
You can assemble yourself and call the renderer:
from pathlib import Path
from phoenix_v4.assembler.assemble_v4 import assemble_book_v4

    voice_id_override=None,  # or a raw ElevenLabs voice ID
)
