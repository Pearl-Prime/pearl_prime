# phoenix_v3/validator/story_slot_validator.py
from dataclasses import dataclass
from typing import Dict, List, Set, Tuple, Any
import re

class PlanValidationError(ValueError):
    pass

@dataclass(frozen=True)
class StorySlotPolicy:
    allowed_layers: Set[str]
    allowed_positions: Set[str]
    position_patterns: Dict[str, re.Pattern]
    global_limits: Dict[str, Any]
    persona_budgets: Dict[str, Any]
    chapter_rules: Dict[str, Any]

def _is_valid_position(pos: str, policy: StorySlotPolicy) -> bool:
    # direct tokens
    if pos in policy.allowed_positions:
        return True
    # patterns
    if policy.position_patterns["after_para_N"].match(pos):
        return True
    if policy.position_patterns["after_section_N"].match(pos):
        return True
    if policy.position_patterns["socket"].match(pos):
        return True
    return False

def _count_words(text: str) -> int:
    return len([w for w in text.strip().split() if w])

def validate_story_slots(
    plan: Dict[str, Any],
    persona_id: str,
    chapter_count: int,
    story_bank_index: Set[str],              # set(story_id)
    load_story_layer_text,                   # fn(story_id, layer) -> str
    policy: StorySlotPolicy,
) -> None:
    """
    Fail-hard validator for plan.yaml story slots.
    - No selection happens here. Only validates plan references against bank + policy.
    """

    # -------------------------
    # Budget resolution
    # -------------------------
    if persona_id not in policy.persona_budgets:
        raise PlanValidationError(f"Missing persona budget in story_slot_policy: {persona_id}")

    pb = policy.persona_budgets[persona_id]
    min_book = int(pb["min_stories_per_book"])
    max_book = int(pb["max_stories_per_book"])
    per_chapter_cap = int(pb.get("per_chapter_cap", policy.global_limits["max_stories_per_chapter"]))

    # -------------------------
    # Iterate chapters
    # -------------------------
    chapters: List[Dict[str, Any]] = plan.get("chapters", [])
    if len(chapters) != chapter_count:
        raise PlanValidationError(f"Plan chapter_count mismatch: expected {chapter_count}, got {len(chapters)}")

    seen_story_ids: Set[str] = set()
    injections_total = 0
    unique_total = 0
    layer_counts = {"L1": 0, "L2": 0, "L3": 0}

    prev_chapter_had_story = False

    for ch in chapters:
        ch_num = str(ch.get("chapter"))
        stories = ch.get("stories", [])

        # Allow missing/empty
        if stories is None:
            stories = []
        if not isinstance(stories, list):
            raise PlanValidationError(f"chapters[{ch_num}].stories must be a list")

        # chapter overrides
        chapter_rule = policy.chapter_rules.get(ch_num, {})
        ch_max = int(chapter_rule.get("max_stories", per_chapter_cap))

        if len(stories) > ch_max:
            raise PlanValidationError(
                f"Chapter {ch_num} has {len(stories)} stories, exceeds cap {ch_max}"
            )

        # anti-adjacent stacking (optional global rule)
        if policy.global_limits.get("disallow_adjacent_story_slots", False):
            if prev_chapter_had_story and len(stories) > 0:
                raise PlanValidationError(
                    f"Adjacent story slots not allowed: chapter {int(ch_num)-1} and {ch_num} both include stories"
                )

        if len(stories) > 0:
            prev_chapter_had_story = True
        else:
            prev_chapter_had_story = False

        # validate each story slot
        for i, s in enumerate(stories):
            if not isinstance(s, dict):
                raise PlanValidationError(f"Chapter {ch_num} story[{i}] must be an object")

            story_id = s.get("id")
            layer = s.get("layer")
            position = s.get("position")

            if not story_id or not isinstance(story_id, str):
                raise PlanValidationError(f"Chapter {ch_num} story[{i}] missing string id")
            if not layer or not isinstance(layer, str):
                raise PlanValidationError(f"Chapter {ch_num} story[{i}] missing string layer")
            if not position or not isinstance(position, str):
                raise PlanValidationError(f"Chapter {ch_num} story[{i}] missing string position")

            if layer not in policy.allowed_layers:
                raise PlanValidationError(
                    f"Invalid layer '{layer}' for story {story_id} in chapter {ch_num}. "
                    f"Allowed: {sorted(policy.allowed_layers)}"
                )

            if not _is_valid_position(position, policy):
                raise PlanValidationError(
                    f"Invalid position '{position}' for story {story_id} in chapter {ch_num}"
                )

            if story_id not in story_bank_index:
                raise PlanValidationError(f"Story not found in bank: {story_id} (chapter {ch_num})")

            # duplicates (fail hard)
            if not policy.global_limits.get("allow_duplicate_story_ids_in_book", False):
                if story_id in seen_story_ids:
                    raise PlanValidationError(
                        f"Duplicate story_id not allowed: {story_id} reused in chapter {ch_num}"
                    )

            # load layer text to validate size bounds + existence
            text = load_story_layer_text(story_id, layer)
            wc = _count_words(text)
            if wc < int(policy.global_limits["min_words_per_story"]):
                raise PlanValidationError(
                    f"Story too short: {story_id}:{layer} has {wc} words (min {policy.global_limits['min_words_per_story']})"
                )
            if wc > int(policy.global_limits["max_words_per_story"]):
                raise PlanValidationError(
                    f"Story too long: {story_id}:{layer} has {wc} words (max {policy.global_limits['max_words_per_story']})"
                )

            # accounting
            injections_total += 1
            if story_id not in seen_story_ids:
                seen_story_ids.add(story_id)
                unique_total += 1
            layer_counts[layer] += 1

    # -------------------------
    # Global totals enforcement
    # -------------------------
    if unique_total > int(policy.global_limits["max_unique_stories_per_book"]):
        raise PlanValidationError(
            f"Too many unique stories: {unique_total} > {policy.global_limits['max_unique_stories_per_book']}"
        )

    if injections_total > int(policy.global_limits["max_total_story_injections_per_book"]):
        raise PlanValidationError(
            f"Too many story injections: {injections_total} > {policy.global_limits['max_total_story_injections_per_book']}"
        )

    # Persona min/max budget
    if unique_total < min_book:
        raise PlanValidationError(f"Too few stories for persona {persona_id}: {unique_total} < {min_book}")
    if unique_total > max_book:
        raise PlanValidationError(f"Too many stories for persona {persona_id}: {unique_total} > {max_book}")

    # Layer mix caps
    layer_mix_max = pb.get("layer_mix_max", {})
    for lyr, cap in layer_mix_max.items():
        if layer_counts.get(lyr, 0) > int(cap):
            raise PlanValidationError(
                f"Layer mix exceeded for persona {persona_id}: {lyr} count {layer_counts[lyr]} > {cap}"
            )

Key point: this validator loads the story layer text (lookup-only) to enforce min/max word bounds and to guarantee the ID+layer exists. That keeps the plan honest and prevents “story placeholders” from slipping through.

# phoenix_v4/assembler/assemble_v4.py

class AssemblerV4:
    """V4 section-first assembler."""
    
    def assemble_chapter(self, chapter_plan: ChapterPlanV4, book_plan: BookPlanV4) -> str:
        """
        Assemble chapter using section-first approach.
        """
        sections_content = []
        
        for section_plan in chapter_plan.sections:
            # Select variant deterministically based on book_seed
            variant = self._select_variant(section_plan, book_plan.book_seed)
            
            # Assemble section content
            section_content = self._assemble_section(section_plan, variant, book_plan)
            
            sections_content.append(section_content)
        
        return "\n\n".join(sections_content)
    
    def _select_variant(self, section_plan: SectionPlan, book_seed: int) -> VariantPlan:
        """
        Deterministic variant selection based on book_seed.
        Same seed → same variant selection.
        """
        # Use hash of (book_seed, chapter_num, section_num) to select variant
        hash_value = hash((book_seed, section_plan.section))
        variant_index = hash_value % len(section_plan.variants)
        return section_plan.variants[variant_index]
    
    def _assemble_section(self, section_plan: SectionPlan, variant: VariantPlan, book_plan: BookPlanV4) -> str:
        """
        Assemble a single section.
        """
        # Load section template
        content = self.libraries.load_section(variant.content_id)
        
        # If scene, load location-aware scene
        if section_plan.type == "SCENE" and variant.scene_id:
            scene_content = self.libraries.load_scene_v4(
                variant.scene_id,
                book_plan.persona,
                book_plan.topic,
                variant.location_profile
            )
            content = self._inject_scene(content, scene_content)
        
        # If story-eligible and story present, inject story
        if section_plan.story_eligible and chapter_plan.story_plan:
            story_content = self.libraries.load_story_v4(
                chapter_plan.story_plan.story_id,
                chapter_plan.story_plan.layer
            )
            content = self._inject_story(content, story_content)
        
        # Character substitution
        content = self._substitute_character(content, book_plan.character_id)
        
        # Location hydration (if location_aware)
        if section_plan.location_aware:
            content = self.location_hydrator.hydrate(content, variant.location_profile)
        
        return content
