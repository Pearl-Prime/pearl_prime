        """
        Validate that text maintains wisdom voice.
        Remove any forbidden patterns.
        """

            text = self._remove_first_person(text)
            
        if self.config.forbid_practitioner_voice:

        if self.config.forbid_comparison_language:
            text = self._remove_comparison_language(text)

    def _remove_first_person(self, text: str) -> str:
        """Remove any first-person language that leaked through"""
        
        # This should never trigger if templates are correct,
        # but provides safety net
        
        forbidden_patterns = [
            ("I struggled with", "People struggle with"),
            ("I used to believe", "The common belief is"),
            ("I learned", "What becomes clear"),
            ("My experience", "What often happens"),
            ("I've seen", "What shows up consistently"),
            ("I found", "What works"),
            ("When I", "When people"),
        ]
        
        for old, new in forbidden_patterns:
            text = text.replace(old, new)
            
        return text
    
    def _remove_practitioner_voice(self, text: str) -> str:
        """Remove practitioner-framed language"""
        
        forbidden_patterns = [
            ("I was helping someone", "There's a pattern where people"),
            ("my client", "someone"),
            ("in my practice", "consistently"),
            ("I worked with someone", "People often experience"),
            ("someone I was working with", "someone"),
        ]
        
        for old, new in forbidden_patterns:
            text = text.replace(old, new)
            
        return text
    
    def _remove_comparison_language(self, text: str) -> str:
        """Remove language that creates comparison"""
        
        forbidden_patterns = [
            ("If I can", "This is"),
            ("you can too", "possible"),
            ("I did it", "it works"),
            ("when I finally", "when the shift happens"),
        ]
        
        for old, new in forbidden_patterns:
            text = text.replace(old, new)
            
        return text

- ✅ Body learns before mind explains
- ✅ Points stated post-exercise (when nervous system regulated)
- ✅ Reinforced through story and application

### Success Criteria

✅ Chapter points appear in three forms (seed/experience/name) not one  
✅ Seed appears before tension peak (implicit recognition)  
✅ Name appears post-exercise (explicit integration)  
✅ Punctuation creates emphasis (no SSML pauses needed)  
✅ Chapter points reinforced through story/carry sentence  
✅ System generates content automatically (not manual editing)

## Architecture Overview

### Current State

Chapter structure:
├─ Validation ("You know what it's like...")
├─ Embodiment (physical sensations)
├─ Tension Peak (daily struggles)
├─ {EXERCISE_SLOT}
├─ Brief integration ("You just mapped your body...")  ← Too quick
└─ Stories + Close

Problem: Chapter point stated once, quickly, then moved on.

### Target State

Enhanced chapter structure:
├─ Validation
├─ SEED (implicit recognition)  ← NEW
├─ Embodiment + Tension Peak
├─ {EXERCISE_SLOT}
├─ NAME (explicit integration)  ← ENHANCED
│   ├─ Core statement
│   ├─ What it's NOT
│   ├─ Practical application
│   └─ Mechanism reinforcement
├─ Recognition story (reinforces point)  ← ENHANCED
└─ Carry sentence (echoes point)  ← ENHANCED

Result: Chapter point appears 3 times in different forms

### Key Design Principles

1. **Seed is implicit** (never states the skill/solution directly)
2. **Experience is somatic** (body learns through exercise)
3. **Name is explicit** (states point clearly post-exercise)
4. **Punctuation creates emphasis** (no SSML pauses)
5. **No verbatim repetition** (same point, different angles)

## The Three-Beat Pattern

### Beat 1: Seed (Implicit Recognition)

**Purpose:** Plant emotional soil without stating the solution

**Placement:** After validation, before embodied struggle

**Structure:**
[Observation about body/nervous system behavior]

[Validation that this isn't a choice]

**Example:**
Your body didn't wait for permission to react. 
It just started bracing. Tightening. Preparing for 
impact even when nothing was coming.

That's not a choice you made. That's your nervous 
system doing what nervous systems do when they've 
been running on empty for too long.

**What this does:**
- Makes listener recognize: "yes, that's happening to me"
- Doesn't name the skill or solution yet
- Creates readiness for the exercise
- Feels like validation, not instruction

**Rules:**
- ❌ Never state the chapter point explicitly
- ❌ Never say "you need to..."
- ❌ Never reference the exercise that's coming
- ✅ Describe what's already happening
- ✅ Validate that it makes sense
- ✅ Keep to 2-4 sentences

---

### Beat 2: Experience (Exercise)

**Purpose:** Body learns the skill somatically

**Placement:** At tension peak

**Structure:**
{EXERCISE_SLOT}

**What this does:**
- Provides somatic relief at peak tension
- Body learns before mind explains
- Creates receptive state for integration

**No changes needed** - this is existing {EXERCISE_SLOT}

---

### Beat 3: Name (Integration)

**Purpose:** State chapter point explicitly when listener is receptive

**Placement:** Immediately after exercise

**Structure:**
[Core statement of chapter point]

[What it's NOT - prevents confusion]

[Practical application - makes it real]

[Mechanism reinforcement - builds trust]

**Example:**
You just practiced the core skill of this chapter.

Building a fast, repeatable way to settle your body 
when burnout spikes.

Not by fixing what's wrong. Not by solving what's 
overwhelming you. Just by giving your nervous system 
a way to stand down without needing everything to be 
okay first.

That body scan? You can return to it anytime. In a 
meeting when your chest tightens. On the subway when 
panic starts building. At 3am when your mind won't 
stop spinning.

Your body knows this pattern now.

The more you practice it, the faster it responds. 
Not because you're getting better at controlling 
anxiety. Because your nervous system is learning it 
can settle without waiting for permission.

**What this does:**
- States point clearly (first paragraph)
- Prevents misinterpretation (second paragraph)
- Makes it concrete (third paragraph)
- Explains mechanism (fourth paragraph)

**Rules:**
- ✅ Start with "You just practiced..."
- ✅ State chapter point in first 1-2 sentences
- ✅ Include "Not by... Just by..." contrast
- ✅ Give 2-3 concrete scenarios where skill applies
- ✅ End with mechanism explanation
- ❌ Don't make it feel like a lecture
- ❌ Don't repeat chapter point verbatim multiple times

## Punctuation-Based Emphasis

### The Challenge

TTS from plain text - can't use SSML <break> tags.

**Must create emphasis through:**
- Sentence structure
- Paragraph breaks
- Punctuation
- Strategic isolation

### Techniques

#### 1. Paragraph Breaks = Natural Pauses

Long sentence that builds context and gives the 
listener time to process.

New paragraph creates a natural pause.

Another sentence continues the thought.

**TTS behavior:** Pause at paragraph break longer than sentence break.

---

#### 2. Short Sentences = Emphasis

Here's what you just learned: you can settle your 
body without solving anything first.

That's the skill.

**The isolated "That's the skill." lands harder** because it's alone.

---

#### 3. Commas = Micro-Pauses

Your body knows this pattern now, and the more you 
practice it, the faster it responds.

**Commas create rhythm** without breaking sentence flow.

---

#### 4. Strategic Repetition Without Verbatim

Say it three ways, never exact same words:

1. "Building a fast, repeatable way to settle your body"
2. "Giving your nervous system a way to stand down"
3. "Your body knows this pattern now"

**All mean the same thing** (the chapter skill) but phrased differently.

---

#### 5. Question → Answer Structure

What does this mean?

It means your nervous system can learn to settle 
without needing everything fixed first.

**Question creates anticipation**, answer delivers payoff.

### Step 1: Create Chapter Point Enhancement Templates

**File:** SOURCE_OF_TRUTH/templates/chapter_point_enhancement.py (NEW)

python
#!/usr/bin/env python3
"""
Chapter Point Enhancement Templates
====================================

Generates Seed → Experience → Name content for chapter points.

STRUCTURE:
- Seed: Implicit recognition (before exercise)
- Experience: {EXERCISE_SLOT} (no changes)
- Name: Explicit integration (after exercise)
"""

from typing import Dict, Optional
from dataclasses import dataclass


@dataclass
class ChapterPointEnhancement:
    """
    Enhanced content for driving chapter points home.
    
    Contains three components:
    - seed: Implicit recognition (2-4 sentences)
    - name_integration: Post-exercise integration (4 paragraphs)
    - story_reinforcement: How to frame recognition story
    - carry_sentence_echo: How to echo point in carry sentence
    """
    
    seed: str
    name_integration: str
    story_reinforcement: str
    carry_sentence_echo: str


class ChapterPointEnhancer:
    """
    Generates enhanced chapter point content.
    
    Takes chapter_point_block and generates:
    1. Seed (implicit recognition)
    2. Name integration (explicit post-exercise)
    3. Story reinforcement hints
    4. Carry sentence echoes
    """
    
    def generate_enhancement(
        self,
        chapter_point: str,
        topic: str,
        exercise_name: str,
        chapter_num: int
    ) -> ChapterPointEnhancement:
        """
        Generate all enhancement components.
        
        Args:
            chapter_point: The core chapter point (e.g., "Build a fast, repeatable way...")
            topic: Book topic (e.g., "burnout", "anxiety")
            exercise_name: Name of exercise (e.g., "body scan", "box breathing")
            chapter_num: Chapter number (1-12)
            
        Returns:
            ChapterPointEnhancement with all components
        """
        
        # Generate seed
        seed = self._generate_seed(chapter_point, topic)
        
        # Generate name integration
        name_integration = self._generate_name_integration(
            chapter_point, topic, exercise_name
        )
        
        # Generate story reinforcement
        story_reinforcement = self._generate_story_hint(chapter_point)
        
        # Generate carry sentence echo
        carry_sentence_echo = self._generate_carry_echo(chapter_point)
        
        return ChapterPointEnhancement(
            seed=seed,
            name_integration=name_integration,
            story_reinforcement=story_reinforcement,
            carry_sentence_echo=carry_sentence_echo
        )
    
    def _generate_seed(self, chapter_point: str, topic: str) -> str:
        """
        Generate implicit recognition seed.
        
        CRITICAL: Never state the chapter point explicitly.
        Only describe what's already happening.
        """
        
        # Extract the core concept from chapter point
        # Example: "Build a fast way to settle" → focus on "body always activated"
        
        # Template varies by chapter point theme
        if "settle" in chapter_point.lower() or "calm" in chapter_point.lower():
            return self._seed_template_settling(topic)
        elif "notice" in chapter_point.lower() or "recognize" in chapter_point.lower():
            return self._seed_template_noticing(topic)
        elif "regulate" in chapter_point.lower():
            return self._seed_template_regulating(topic)
        else:
            return self._seed_template_generic(topic)
    
    def _seed_template_settling(self, topic: str) -> str:
        """Seed for settling/calming skills"""
        return f"""Your body didn't wait for permission to react. 
It just started bracing. Tightening. Preparing for 
impact even when nothing was coming.

That's not a choice you made. That's your nervous 
system doing what nervous systems do when they've 
been running on empty for too long."""
    
    def _seed_template_noticing(self, topic: str) -> str:
        """Seed for noticing/awareness skills"""
        return f"""Your body has been trying to tell you something.

Not through thoughts or words. Through sensations. 
Tightness. Tension. That knot in your stomach that 
shows up before you've even named what you're worried about.

That's information, not noise."""
    
    def _seed_template_regulating(self, topic: str) -> str:
        """Seed for regulation skills"""
        return f"""When {topic} spikes, your body doesn't wait for 
your brain to catch up.

Heart rate increases. Breath gets shallow. Muscles 
tense. All of that happens before you've consciously 
registered what's happening.

That's your nervous system doing its job. The question 
isn't whether it responds. It's whether you can work 
with that response."""
    
    def _seed_template_generic(self, topic: str) -> str:
        """Generic seed template"""
        return f"""There's a pattern with {topic} that most people 
don't notice at first.

It builds quietly. Day by day. Until one day it's 
not quiet anymore.

That progression isn't random. It's predictable."""
    
    def _generate_name_integration(
        self,
        chapter_point: str,
        topic: str,
        exercise_name: str
    ) -> str:
        """
        Generate post-exercise integration block.
        
        STRUCTURE:
        1. "You just practiced [CHAPTER POINT]"
        2. Not by... Just by... (prevents confusion)
        3. Concrete application scenarios
        4. Mechanism explanation
        """
        
        # Extract key action from chapter point
        # Example: "Build a fast way to settle" → "settling your body"
        action = self._extract_action_from_point(chapter_point)
        
        # Build the integration
        return f"""You just practiced the core skill of this chapter.

{chapter_point}

Not by fixing {topic}. Not by solving what's overwhelming you. 
Just by giving your nervous system a way to {action} without 
needing everything to be okay first.

That {exercise_name}? You can return to it anytime. In a 
meeting when tension starts building. On the subway when 
your mind starts racing. At 3am when you can't shut down.

Your body knows this pattern now.

The more you practice it, the faster it responds. Not because 
you're getting better at controlling {topic}. Because your 
nervous system is learning it can {action} without waiting 
for permission."""
    
    def _extract_action_from_point(self, chapter_point: str) -> str:
        """
        Extract the core action from chapter point.
        
        Examples:
        - "Build a way to settle" → "settle"
        - "Learn to notice signals" → "notice early signals"
        - "Regulate breath" → "regulate"
        """
        
        # Simple keyword extraction
        if "settle" in chapter_point.lower():
            return "settle"
        elif "notice" in chapter_point.lower():
            return "notice what's happening"
        elif "regulate" in chapter_point.lower():
            return "regulate"
        elif "recognize" in chapter_point.lower():
            return "recognize the pattern"
        else:
            return "work with your nervous system"
    
    def _generate_story_hint(self, chapter_point: str) -> str:
        """
        Generate hint for how recognition story should reinforce point.
        
        This is metadata for story selection, not actual story text.
        """
        
        return f"Story should show someone using this skill: {chapter_point}. " \
               f"Focus on the moment they apply it, not their entire journey."
    
    def _generate_carry_echo(self, chapter_point: str) -> str:
        """
        Generate a version of chapter point for carry sentence.
        
        Should echo the point but not repeat it verbatim.
        """
        
        # Rephrase the chapter point
        if "build" in chapter_point.lower():
            return chapter_point.replace("Build", "You have")
        elif "learn" in chapter_point.lower():
            return chapter_point.replace("Learn", "You know how")
        else:
            return f"Remember: {chapter_point.lower()}"


# Main entry point
def generate_chapter_point_enhancement(
    chapter_point: str,
    topic: str,
    exercise_name: str,
    chapter_num: int
) -> ChapterPointEnhancement:
    """
    Generate complete chapter point enhancement.
    
    Usage:
        enhancement = generate_chapter_point_enhancement(
            chapter_point="Build a fast, repeatable way to settle your body when burnout spikes",
            topic="burnout",
            exercise_name="body scan",
            chapter_num=1
        )
        
        # Use in chapter template
        seed = enhancement.seed
        name_integration = enhancement.name_integration
    """
    
    enhancer = ChapterPointEnhancer()
    
    return enhancer.generate_enhancement(
        chapter_point=chapter_point,
        topic=topic,
        exercise_name=exercise_name,
        chapter_num=chapter_num
    )

#!/usr/bin/env python3
"""
Seed Template Variations
========================

Multiple seed templates for variety across chapters.

All follow same rule: IMPLICIT recognition, never explicit instruction.
"""

from typing import List, Dict
import random


class SeedTemplates:
    """
    Collection of seed templates.
    
    Each template creates implicit recognition without
    stating the chapter point or solution.
    """
    
    # Templates for "settling/calming" skills
    SETTLING_SEEDS = [
        """Your body didn't wait for permission to react. 
It just started bracing. Tightening. Preparing for 
impact even when nothing was coming.

That's not a choice you made. That's your nervous 
system doing what nervous systems do when they've 
been running on empty for too long.""",
        
        """There's a moment before {topic} fully hits when 
your body already knows.

Your jaw tightens. Your shoulders creep up. Your 
breath gets shallow. All of that happens before 
you've consciously registered what's coming.

That's your early warning system. It's been running 
this whole time.""",
        
        """When your nervous system is overwhelmed, it 
doesn't shut down. It speeds up.

More thoughts. Faster heartbeat. Tighter muscles. 
Everything accelerates when what you actually need 
is the opposite.

That's the paradox of {topic}."""
    ]
    
    # Templates for "noticing/awareness" skills
    NOTICING_SEEDS = [
        """Your body has been trying to tell you something.

Not through thoughts or words. Through sensations. 
Tightness. Tension. That knot in your stomach that 
shows up before you've even named what you're worried about.

That's information, not noise.""",
        
        """Most people miss the early signs of {topic} 
because they're looking for the big moments.

But {topic} doesn't announce itself. It whispers 
first. A slight tension. A faster heartbeat. A 
shallow breath.

By the time it's loud, it's already been building.""",
        
        """{topic} has a signature in your body.

It shows up the same way every time. Same tension 
pattern. Same sensations. Same physical response.

You might not have noticed the pattern yet. But 
your body has."""
    ]
    
    # Templates for "regulation/breath" skills
    REGULATION_SEEDS = [
        """When {topic} spikes, your body doesn't wait for 
your brain to catch up.

Heart rate increases. Breath gets shallow. Muscles 
tense. All of that happens before you've consciously 
registered what's happening.

That's your nervous system doing its job. The question 
isn't whether it responds. It's whether you can work 
with that response.""",
        
        """Your breath is the first thing that changes when 
{topic} shows up.

It gets shorter. Faster. Shallower. You might not 
notice it consciously, but your body does.

That shift happens automatically. What comes next 
doesn't have to.""",
        
        """There's a feedback loop between your breath and 
your nervous system.

Shallow breath signals threat. Deep breath signals 
safety. Your body responds to whichever signal it 
receives.

Right now, that loop is running automatically."""
    ]
    
    @classmethod
    def get_seed(
        cls,
        skill_type: str,
        topic: str,
        variation: int = 0
    ) -> str:
        """
        Get a seed template.
        
        Args:
            skill_type: "settling", "noticing", or "regulation"
            topic: Book topic (for {topic} substitution)
            variation: Which variation to use (0, 1, or 2), or random if -1
            
        Returns:
            Seed text with {topic} filled in
        """
        
        # Select template collection
        if skill_type == "settling":
            templates = cls.SETTLING_SEEDS
        elif skill_type == "noticing":
            templates = cls.NOTICING_SEEDS
        elif skill_type == "regulation":
            templates = cls.REGULATION_SEEDS
        else:
            # Default to settling
            templates = cls.SETTLING_SEEDS
        
        # Select variation
        if variation == -1:
            template = random.choice(templates)
        else:
            template = templates[variation % len(templates)]
        
        # Fill in topic
        return template.format(topic=topic)

## Migration Guide

### For Existing Books

**Backward Compatibility:**

Existing chapters continue to work - enhancement is additive:
- Old chapters without seed/enhanced integration still function
- New chapters get enhanced structure
- No breaking changes
