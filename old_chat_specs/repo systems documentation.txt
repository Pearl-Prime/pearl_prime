python
# validator/exhaustion_validator.py (updated)

from typing import Dict, List, Optional
from .models import *

class ExhaustionValidator:
    def __init__(self, inventory_path: str, template_path: str):
        self.insights: Dict[str, Insight] = {}
        self.beats: List[Beat] = []
        self.insight_states: Dict[str, InsightState] = {}
        self.results: List[ValidationResult] = []
        self.mode: ValidatorMode = ValidatorMode.STRICT
        
        # Governor tracking
        self.explain_counts: Dict[str, Dict[int, int]] = {}
        self.demonstrate_counts: Dict[str, Dict[int, int]] = {}
        
        # Governor config (defaults)
        self.max_explain_per_chapter = 1
        self.max_demonstrate_per_chapter = 3
        self.strict_demonstrate = False
        
        self._load_inventory(inventory_path)
        self._load_template(template_path)
        self._initialize_states()
    
    def _load_inventory(self, path: str):
        with open(path, 'r') as f:
            data = yaml.safe_load(f)
        
        meta = data.get('meta', {})
        self.mode = ValidatorMode(meta.get('validator_mode', 'strict'))
        self.max_explain_per_chapter = meta.get('max_explain_per_insight_per_chapter', 1)
        self.max_demonstrate_per_chapter = meta.get('max_demonstrate_per_insight_per_chapter', 3)
        self.strict_demonstrate = meta.get('strict_demonstrate', False)
        
        for insight_id, config in data.get('insights', {}).items():
            self.insights[insight_id] = Insight(
                id=insight_id,
                owning_chapter=config['owning_chapter'],
                core_idea=config['core_idea'],
                aliases=config.get('aliases', []),
                exhaustion_point=config.get('exhaustion_point', '')
            )
    
    def _validate_beat(self, beat: Beat, insight_id: str) -> ValidationResult:
        if insight_id not in self.insights:
            return ValidationResult(
                beat=beat,
                insight_id=insight_id,
                status="WARN",
                message=f"Unknown insight ID: {insight_id}",
                insight_state=InsightState.UNINTRODUCED
            )
        
        insight = self.insights[insight_id]
        state = self.insight_states[insight_id]
        role = beat.insight_role
        
        # NEUTRAL always passes
        if role == InsightRole.NEUTRAL:
            return ValidationResult(beat=beat, insight_id=insight_id, status="PASS",
                                    message="Neutral role", insight_state=state)
        
        # EXPLAIN rules
        if role == InsightRole.EXPLAIN:
            # Check exhaustion (cross-chapter)
            if state == InsightState.EXHAUSTED:
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status="FAIL",
                    message=f"Cannot EXPLAIN exhausted insight '{insight_id}'",
                    insight_state=state
                )
            
            # Check premature (before owning chapter)
            if state == InsightState.UNINTRODUCED and beat.chapter != insight.owning_chapter:
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status="FAIL",
                    message=f"Cannot EXPLAIN '{insight_id}' before Ch{insight.owning_chapter}",
                    insight_state=state
                )
            
            # Check in-chapter cap (Governor 1)
            chapter_counts = self.explain_counts.setdefault(insight_id, {})
            current_count = chapter_counts.get(beat.chapter, 0)
            
            if current_count >= self.max_explain_per_chapter:
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status="FAIL",
                    message=f"EXPLAIN cap exceeded: '{insight_id}' already explained "
                            f"{current_count}x in Ch{beat.chapter} (max: {self.max_explain_per_chapter})",
                    insight_state=state
                )
            
            # Valid EXPLAIN - increment count
            chapter_counts[beat.chapter] = current_count + 1
            return ValidationResult(
                beat=beat, insight_id=insight_id, status="PASS",
                message=f"Valid EXPLAIN #{current_count + 1} of '{insight_id}'",
                insight_state=state
            )
        
        # DEMONSTRATE rules
        if role == InsightRole.DEMONSTRATE:
            # Check saturation cap (Governor 2)
            chapter_counts = self.demonstrate_counts.setdefault(insight_id, {})
            current_count = chapter_counts.get(beat.chapter, 0)
            
            if current_count >= self.max_demonstrate_per_chapter:
                status = "FAIL" if self.strict_demonstrate else "WARN"
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status=status,
                    message=f"DEMONSTRATE saturation: '{insight_id}' has "
                            f"{current_count} demos in Ch{beat.chapter} (max: {self.max_demonstrate_per_chapter})",
                    insight_state=state
                )
            
            # Check demonstrate before explain (soft warning)
            if state == InsightState.UNINTRODUCED:
                # Still increment, but warn
                chapter_counts[beat.chapter] = current_count + 1
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status="WARN",
                    message=f"DEMONSTRATE before EXPLAIN for '{insight_id}'",
                    insight_state=state
                )
            
            # Valid DEMONSTRATE
            chapter_counts[beat.chapter] = current_count + 1
            return ValidationResult(
                beat=beat, insight_id=insight_id, status="PASS",
                message=f"Valid DEMONSTRATE #{current_count + 1} of '{insight_id}'",
                insight_state=state
            )
        
        # REFERENCE rules
        if role == InsightRole.REFERENCE:
            if state == InsightState.UNINTRODUCED:
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status="FAIL",
                    message=f"Cannot REFERENCE unintroduced insight '{insight_id}'",
                    insight_state=state
                )
            return ValidationResult(
                beat=beat, insight_id=insight_id, status="PASS",
                message=f"Valid REFERENCE to '{insight_id}'",
                insight_state=state
            )
        
        # ASSUME rules
        if role == InsightRole.ASSUME:
            if state == InsightState.UNINTRODUCED:
                return ValidationResult(
                    beat=beat, insight_id=insight_id, status="WARN",
                    message=f"ASSUME before EXPLAIN for '{insight_id}'",
                    insight_state=state
                )
            return ValidationResult(
                beat=beat, insight_id=insight_id, status="PASS",
                message=f"Valid ASSUME of '{insight_id}'",
                insight_state=state
            )
        
        return ValidationResult(
            beat=beat, insight_id=insight_id, status="PASS",
            message="No rule matched", insight_state=state
        )
