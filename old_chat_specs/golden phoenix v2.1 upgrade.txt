def count_stories_by_role(repo_root: Path, topic: str, teacher_id: str) -> Dict[str, int]:
    """
    Reads teacher_bank stories for this topic and counts by role.
    Stories without a role (or with role not in STORY_ROLES) are not counted toward any role.
    """
    role_counts: Dict[str, int] = {r: 0 for r in STORY_ROLES}
    banks_dir = repo_root / "SOURCE_OF_TRUTH" / "data" / "teacher_banks" / teacher_id
    stories_dir = banks_dir / "stories"
    if not stories_dir.exists():
        return role_counts
    prefix = f"{teacher_id}_{topic}_"
    for path in stories_dir.glob(f"{prefix}*.json"):
        try:
            data = json.loads(path.read_text(encoding="utf-8"))
            role = (data.get("role") or "").strip().lower()
            if role in STORY_ROLES:
                role_counts[role] = role_counts.get(role, 0) + 1
        except Exception:
            continue
    return role_counts

            skipped += 1
            continue

            body_text, meta_dict = writer(text, parsed)
        except Exception as e:
            rejected += 1
            rejection_reasons["writer_error"] = rejection_reasons.get("writer_error", 0) + 1
            cb(f"Reject (writer): {pack_path.name} — {e}")
            continue

        if not body_text or not meta_dict:
            skipped += 1
            continue

        point = (meta_dict.get("point") or "").strip()
        if not point:
            rejected += 1
            rejection_reasons["AVC-1 missing"] = rejection_reasons.get("AVC-1 missing", 0) + 1
            cb(f"Reject (no point): {pack_path.name}")
            continue

        if _point_already_used_atom(repo_root, persona, topic, role, point):
            cb(f"Skip (point exists): {persona}/{topic}/{role}")
            skipped += 1
            continue

            cb(f"Skip (situation exists): {persona}/{topic}/{role}")
            skipped += 1
            continue

            "text": body_text,
            "point": point,
            "reversal_line": meta_dict.get("reversal_line") or "",
            "stakes": meta_dict.get("stakes") or "",
            "role": role,
        }
        if not lint_story_atom:
            rejection_reasons["lint_unavailable"] = rejection_reasons.get("lint_unavailable", 0) + 1
            rejected += 1
            continue

        if errs:
            rejected += 1
            for e in errs:
                key = getattr(e, "rule", str(e)) if hasattr(e, "rule") else "lint"
                rejection_reasons[key] = rejection_reasons.get(key, 0) + 1
            cb(f"Reject (lint): {pack_path.name} — {errs[0]}")
            continue

        meta_out = {
            "point": point,
            "reversal_line": meta_dict.get("reversal_line") or None,
            "stakes": meta_dict.get("stakes") or None,
            "status": "provisional",
        }
        save_dir.mkdir(parents=True, exist_ok=True)
        (save_dir / body_file).write_text(body_text.strip(), encoding="utf-8")
        if yaml:

                yaml.dump(meta_out, default_flow_style=False, allow_unicode=True),
                encoding="utf-8",
            )
        else:

        written += 1
        cb(f"Written: {persona}/{topic}/{role} {body_file}")

    return {
        "written": written,
        "rejected": rejected,
        "rejection_reasons": rejection_reasons,
        "skipped": skipped,
    }

#!/usr/bin/env python3
"""
Golden Phoenix freeze + validate: no new content.
- RB-1 (role balance) and teacher_bank validator for all teachers (run separately or via Make).
- Lint sample of system atoms (50).
- Confirm fingerprint files exist.

Exit 0 if all pass; 1 otherwise.
"""

from __future__ import annotations

import json
import sys
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parent.parent
if str(REPO_ROOT) not in sys.path:
    sys.path.insert(0, str(REPO_ROOT))


def main() -> int:
    from phoenix_v4.ci.story_atom_lint import lint_story_atom
    import yaml

    atoms_dir = REPO_ROOT / "SOURCE_OF_TRUTH" / "data" / "story_bank" / "story_atoms"
    checked, errors = 0, []
    for txt in list(atoms_dir.rglob("micro/v*.txt"))[:50]:
        meta_path = txt.parent / (txt.stem + "_meta.yaml")
        if not meta_path.exists():
            continue
        body = txt.read_text(encoding="utf-8")
        meta = yaml.safe_load(meta_path.read_text(encoding="utf-8")) or {}
        atom = {
            "text": body,
            "point": meta.get("point") or "",
            "reversal_line": meta.get("reversal_line") or "",
            "stakes": meta.get("stakes") or "",
            "role": meta.get("role") or txt.parent.parent.name,
        }
        errs = lint_story_atom(atom)
        checked += 1
        if errs:
            errors.append((str(txt.relative_to(atoms_dir)), [e.rule for e in errs]))
    print(f"Story-atom lint: {checked} sampled.", "All passed." if not errors else f"{len(errors)} failed.")
    if errors:
        for p, r in errors[:10]:
            print(f"  {p}: {r}")
        return 1

    # Fingerprint presence
    for tid in ("ahjan", "kevin_costner", "sai_maa"):
        p = REPO_ROOT / "SOURCE_OF_TRUTH" / "data" / "teacher_banks" / tid / ".fingerprints.json"
        if not p.exists():
            print(f"Missing teacher_bank fingerprints: {tid}")
            return 1
    pa = REPO_ROOT / "SOURCE_OF_TRUTH" / "data" / "story_bank" / "story_atoms" / ".fingerprints.json"
    if not pa.exists():
        print("Missing system_atoms .fingerprints.json")
        return 1
    print("Fingerprint files present.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
