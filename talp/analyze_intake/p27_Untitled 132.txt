#!/usr/bin/env python3
"""
persona_research.py

Deep research harvester for Phoenix V4 personas + location cues.

Outputs (V4-aligned):
- registry/person:contentReference[oaicite:11]{index=11}bank/location_variables/<locale>.yaml
- data/story_bank/social_patterns/<persona>.yaml
- data/story_bank/pressure_themes/<persona>.yaml
- artifacts/persona_research/report.json

Design goals:
- schema-first JSON extraction (LLM produces JSON only)
- citations preserved (source URLs + retrieval timestamps)
- coarse demographics only (no sensitive inference)
- location cues live in story_bank (never inside atoms)
"""

# Config

DEFAULT_OUT_ROOT = Path("SOURCE_OF_TRUTH")
HTTP_TIMEOUT = 30

GENERATION_SOURCES = [
    # Prefer stable / primary sources; keep a small curated list
    "https://www.census.gov/newsroom/press-releases/2025/birth-cohorts-geographic-mobility-report.html",
    "https://www.pewresearch.org/short-reads/2019/01/17/where-millennials-end-and-generation-z-begins/",
]

LOCALE_SOURCES = {
    "nyc": [

def now_utc_iso() -> str:
    return dt.datetime.now(dt.timezone.utc).isoformat()

def sha256_text(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()

def http_get(url: str) -> str:
    r = requests.get(url, timeout=HTTP_TIMEOUT, headers={"User-Agent": "persona-research/1.0"})
    r.raise_for_status()
    return r.text

def load_yaml(path: Path) -> Any:
    if not path.exists():
        return None
    return yaml.safe_load(path.read_text(encoding="utf-8"))

def dump_yaml(path: Path, obj: Any) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(yaml.safe_dump(obj, sort_keys=False, allow_unicode=True), encoding="utf-8")

def dump_json(path: Path, obj: Any) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(obj, ensure_ascii=False, indent=2), encoding="utf-8")

REQUIRED_PERSONA_KEYS = {

def validate_persona_profile(obj: Dict[str, Any]) -> None:
    missing = REQUIRED_PERSONA_KEYS - set(obj.keys())
    if missing:
        raise ValueError(f"persona_profile missing keys: {sorted(missing)}")

SYSTEM_JSON_ONLY = (

    "Return ONLY valid JSON. No markdown. No commentary."
)

def persona_prompt(persona_id: str, topics: List[str], locale_id: Optional[str], sources: List[Dict[str, str]]) -> str:
    return json.dumps({
        "task": "extract_persona_profile",
        "persona_id": persona_id,
        "topics": topics,
        "locale_id": locale_id,
        "instructions": [
            "Use the provided sources only; do not invent facts.",

            "Return a persona profile that maps how each topic shows up for this persona (signals + triggers).",
            "Do NOT put hardcoded place names into atoms; output location cues as placeholders + cue set references.",
            "Include a 'sources' field with URLs used and short justification per URL."
        ],
        "desired_schema": {
            "persona_id": "string",
            "display_name": "string",
            "generation": {
                "label": "string",
                "birth_years": ["int_start", "int_end_or_null"],

            "demographics": {

                "education_band": "string"

            "sources": [{"url": "string", "used_for": "string"}]

def run(personas: List[str], topics: List[str], locales: List[str], out_root: Path) -> None:
    report: Dict[str, Any] = {
        "run_id": sha256_text(now_utc_iso())[:12],
        "ran_at": now_utc_iso(),
        "personas": [],
        "locales": [],
        "errors": [],
    }

    # 1) Fetch sources (cache optional)
    fetched_pages: Dict[str, str] = {}
    all_urls = set(GENERATION_SOURCES)
    for loc in locales:
        for u in LOCALE_SOURCES.get(loc, []):
            all_urls.add(u)

    for url in sorted(all_urls):
        try:
            fetched_pages[url] = http_get(url)
        except Exception as e:
            report["errors"].append({"stage": "fetch", "url": url, "error": str(e)})

    # 2) Locale cue extraction
    for locale_id in locales:
        src_urls = LOCALE_SOURCES.get(locale_id, [])
        src_pack = [{"url": u, "content": fetched_pages.get(u, "")[:12000]} for u in src_urls]
        if not src_pack:
            continue
        try:
            locale_json = call_llm_json(SYSTEM_JSON_ONLY, locale_cues_prompt(locale_id, src_pack))
            # Write story_bank location variables
            dump_yaml(out_root / "data/story_bank/location_variables" / f"{locale_id}.yaml", locale_json)
            report["locales"].append({"locale_id": locale_id, "sources": [u["url"] for u in locale_json.get("sources", [])]})
        except Exception as e:
            report["errors"].append({"stage": "locale_llm", "locale_id": locale_id, "error": str(e)})

    # 3) Persona extraction
    personas_yaml_path = out_root / "registry/personas.yaml"
    personas_registry = load_yaml(personas_yaml_path) or {"personas": {}}

    gen_src_pack = [{"url": u, "content": fetched_pages.get(u, "")[:12000]} for u in GENERATION_SOURCES]

    for persona_id in personas:
        # optional: infer locale from persona_id naming convention
        locale_id = None
        m = re.search(r"(nyc|la|chicago|taipei|tokyo)", persona_id)
        if m:
            locale_id = m.group(1)

        # assemble sources pack
        src_pack = list(gen_src_pack)
        if locale_id and (out_root / "data/story_bank/location_variables" / f"{locale_id}.yaml").exists():
            # include locale source pages too (for cues)
            for u in LOCALE_SOURCES.get(locale_id, []):
                src_pack.append({"url": u, "content": fetched_pages.get(u, "")[:12000]})

        try:
            persona_json = call_llm_json(SYSTEM_JSON_ONLY, persona_prompt(persona_id, topics, locale_id, src_pack))
            validate_persona_profile(persona_json)

            # Merge into registry
            personas_registry["personas"][persona_id] = persona_json

            # Optionally also emit overlays
            dump_yaml(out_root / "data/story_bank/social_patterns" / f"{persona_id}.yaml",
                      {"persona_id": persona_id, "social_patterns": persona_json.get("topic_fit", {})})
            dump_yaml(out_root / "data/story_bank/pressure_themes" / f"{persona_id}.yaml",
                      {"persona_id": persona_id, "pressure_themes": persona_json.get("topic_fit", {})})

            report["personas"].append({
                "persona_id": persona_id,
                "locale_id": locale_id,
                "sources": [s["url"] for s in persona_json.get("sources", [])]
            })
        except Exception as e:
            report["errors"].append({"stage": "persona_llm", "persona_id": persona_id, "error": str(e)})

    dump_yaml(personas_yaml_path, personas_registry)
    dump_json(out_root / "artifacts/persona_research/report.json", report)

def parse_args() -> argparse.Namespace:
    ap = argparse.ArgumentParser()
    ap.add_argument("--out-root", default=str(DEFAULT_OUT_ROOT))
    ap.add_argument("--personas", nargs="+", required=True)
    ap.add_argument("--topics", nargs="+", required=True)
    ap.add_argument("--locales", nargs="*", default=[])
    return ap.parse_args()
